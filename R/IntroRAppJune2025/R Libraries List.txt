# Example: Using multiple libraries together
library(dplyr)
library(ggplot2)
library(plotly)

plot <- mtcars %>%
  mutate(efficiency = ifelse(mpg > 20, 'High', 'Low')) %>%
  ggplot(aes(x = wt, y = mpg, color = efficiency)) +
  geom_point(size = 3) +
  labs(title = 'Car Weight vs MPG') +
  theme_minimal()

# Make it interactive
ggplotly(plot)
